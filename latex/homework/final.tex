\section{Maybe these would be good final exam problems}

\subsection*{Prove things}

\begin{problem}
    Prove the ``one-step subgroup test:'' Let $H\subseteq G$. If $xy\inv \in H$ for all $x, y\in H$, then $H$ is a subgroup of $G$.
\end{problem}

\begin{problem}
    What does Lagrange's theorem say? Prove it.
\end{problem}

\begin{problem}
    Let $G$ be a group, $N$ a normal subgroup, and $G/N = \{gN \mid g\in G\}$ be the set of left (or, equivalently, right) cosets of $N$.
\end{problem}
